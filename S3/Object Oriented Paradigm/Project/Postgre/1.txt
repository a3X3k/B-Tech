CREATE TABLE AWARDS 
(AWARD_ID VARCHAR (20) PRIMARY KEY,
AWARD_NAME VARCHAR (20),
AWARD_CATEGORY VARCHAR (20),
REWARD_AMOUNT INT,
SPONSORED_BY VARCHAR (20));

CREATE TABLE INVENTOR
(INVENTOR_ID VARCHAR (10) PRIMARY KEY,
NAME VARCHAR (20),
GENDER VARCHAR(20),
EMAIL VARCHAR(20),
PHONE_NO VARCHAR(20),
COUNTRY VARCHAR (20),
JOB_TITLE VARCHAR (20),
QUALIFICATION VARCHAR (20),
SPECIALIZATION VARCHAR (20),
YEAR_OF_EXPERIENCE VARCHAR(20))


CREATE TABLE INVENTION
(INVENTION_ID VARCHAR (20) PRIMARY KEY,
INVENTION_NAME VARCHAR (30),
AWARD_CATEGORY VARCHAR (20),
YEAR_OF_INVENTION INT,
STORY_BEHIND VARCHAR (100),
MARKS NUMERIC )

CREATE TABLE INVENTS
(INVENTOR_ID VARCHAR (10) REFERENCES
INVENTOR(INVENTOR_ID),
INVENTION_ID VARCHAR (10) REFERENCES
INVENTION(INVENTION_ID)
)

CREATE TABLE PANEL
(PANEL_ID VARCHAR (10) PRIMARY KEY,
PANEL_NAME VARCHAR (20),
AWARD_CATEGORY VARCHAR(20))

CREATE TABLE JURY_MEMBERS
(JURY_ID VARCHAR (10) PRIMARY KEY,
JURY_NAME VARCHAR (20),
GENDER VARCHAR(20),
EMAIL VARCHAR(20),
PHONE_NO VARCHAR(20),
COUNTRY VARCHAR(20),
AWARD_CATEGORY VARCHAR(20),
QUALIFICATION VARCHAR (20),
SPECIALIZATION VARCHAR (20),
YEAR_OF_EXPERIENCE VARCHAR(20))

CREATE TABLE PANEL_JURY
(PANEL_ID VARCHAR(20) REFERENCES
PANEL(PANEL_ID),
JURY_ID VARCHAR (10) REFERENCES
JURY_MEMBERS(JURY_ID))

CREATE TABLE LOGIN
(USERNAME VARCHAR(20),
PASSWORD VARCHAR(20),
CATEGORY VARCHAR(20),
ID VARCHAR(20))
